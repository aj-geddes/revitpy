from revitpy import RevitContext

# Query walls with modern ORM syntax
with RevitContext() as context:
    walls = context.elements.of_category('Walls')
    tall_walls = walls.where(lambda w: w.Height > 10.0)

    for wall in tall_walls:
        print(f"Wall: {wall.Name}, Height: {wall.Height}")
from revitpy import RevitContext

with RevitContext() as context:
    # Chain filters with LINQ-style syntax
    results = (context.elements
        .of_category('Rooms')
        .where(lambda r: r.Area > 500)
        .include('Boundaries.Wall')  # Eager load relationships
        .order_by(lambda r: r.Area)
        .take(10)
        .to_list())

    for room in results:
        print(f"{room.Name}: {room.Area} sq ft")
import asyncio
from revitpy import AsyncRevitContext

async def process_elements():
    async with AsyncRevitContext() as context:
        # Process large datasets asynchronously
        elements = await context.elements.where(
            lambda e: e.Category == 'Windows'
        ).to_list_async()

        # Run operations in parallel
        tasks = [update_element(elem) for elem in elements]
        results = await asyncio.gather(*tasks)

        return results

asyncio.run(process_elements())
from revitpy.events import EventManager, event_handler, EventType

@event_handler(EventType.ELEMENT_ADDED)
def on_element_added(event_data):
    """Auto-tag new walls with sequential marks"""
    element = event_data.data['element']

    if element.Category == 'Walls':
        mark = f"W-{element.Id.value}"
        element.set_parameter('Mark', mark)
        print(f"Tagged new wall: {mark}")

# Register and start listening
EventManager.register(on_element_added)
